import React, { useState } from "react";
import { base44 } from "@/api/base44Client";
import { Card, CardContent, CardHeader, CardTitle } from "@/components/ui/card";
import { Button } from "@/components/ui/button";
import { Badge } from "@/components/ui/badge";
import { Upload, FileText, Image as ImageIcon, Loader2, X, AlertCircle, CheckCircle, Sparkles } from "lucide-react";
import { toast } from "sonner";
import { Accordion, AccordionContent, AccordionItem, AccordionTrigger } from "@/components/ui/accordion";

export default function AIDocumentAnalyzer({ user }) {
  const [selectedFile, setSelectedFile] = useState(null);
  const [analysis, setAnalysis] = useState(null);
  const [isAnalyzing, setIsAnalyzing] = useState(false);
  const [analysisType, setAnalysisType] = useState("comprehensive");

  const documentTypes = [
    { value: "comprehensive", label: "üîç Comprehensive Analysis", desc: "Full document analysis with all details" },
    { value: "summary", label: "üìù Summary", desc: "Quick overview and key points" },
    { value: "legal", label: "‚öñÔ∏è Legal Review", desc: "Legal terms, clauses, and risks" },
    { value: "translation", label: "üåê Translation", desc: "Translate document content" },
    { value: "form_data", label: "üìã Extract Form Data", desc: "Extract structured data from forms" }
  ];

  const handleFileSelect = (e) => {
    const file = e.target.files[0];
    if (!file) return;

    if (file.size > 10 * 1024 * 1024) {
      toast.error("File must be less than 10MB");
      return;
    }

    setSelectedFile(file);
    setAnalysis(null);
  };

  const analyzeDocument = async () => {
    if (!selectedFile) {
      toast.error("Please select a file");
      return;
    }

    if (!user) {
      toast.error("Please login to use this feature");
      base44.auth.redirectToLogin();
      return;
    }

    setIsAnalyzing(true);

    try {
      const { file_url } = await base44.integrations.Core.UploadFile({ file: selectedFile });

      let prompt = "";
      let schema = {};

      switch (analysisType) {
        case "comprehensive":
          prompt = `Analyze this document comprehensively and provide:
1. Document type and purpose
2. Key information and main points
3. Important dates, numbers, and entities
4. Overall summary
5. Any potential issues or concerns`;
          schema = {
            type: "object",
            properties: {
              document_type: { type: "string" },
              purpose: { type: "string" },
              key_points: { type: "array", items: { type: "string" } },
              important_details: { type: "object" },
              summary: { type: "string" },
              concerns: { type: "array", items: { type: "string" } }
            }
          };
          break;

        case "summary":
          prompt = `Provide a concise summary of this document including main topic, key points, and conclusion.`;
          schema = {
            type: "object",
            properties: {
              main_topic: { type: "string" },
              key_points: { type: "array", items: { type: "string" } },
              conclusion: { type: "string" }
            }
          };
          break;

        case "legal":
          prompt = `Analyze this document from a legal perspective:
1. Identify all legal terms and clauses
2. Highlight important rights and obligations
3. Point out potential risks or red flags
4. Suggest what to be careful about`;
          schema = {
            type: "object",
            properties: {
              legal_terms: { type: "array", items: { type: "string" } },
              rights_obligations: { type: "array", items: { type: "string" } },
              risks: { type: "array", items: { type: "string" } },
              recommendations: { type: "string" }
            }
          };
          break;

        case "translation":
          prompt = `Translate the content of this document to English and provide the translated text.`;
          schema = {
            type: "object",
            properties: {
              original_language: { type: "string" },
              translated_text: { type: "string" }
            }
          };
          break;

        case "form_data":
          prompt = `Extract all form fields and their values from this document in a structured format.`;
          schema = {
            type: "object",
            properties: {
              form_name: { type: "string" },
              fields: { type: "array", items: { type: "object" } }
            }
          };
          break;
      }

      const result = await base44.integrations.Core.InvokeLLM({
        prompt,
        file_urls: [file_url],
        response_json_schema: schema
      });

      setAnalysis(result);
      toast.success("Document analyzed successfully!");
    } catch (error) {
      console.error("Analysis error:", error);
      toast.error("Failed to analyze document");
    } finally {
      setIsAnalyzing(false);
    }
  };

  const reset = () => {
    setSelectedFile(null);
    setAnalysis(null);
  };

  return (
    <div className="space-y-6">
      {!analysis ? (
        <Card>
          <CardHeader>
            <CardTitle className="flex items-center gap-2">
              <Sparkles className="w-5 h-5 text-purple-600" />
              AI Document Analyzer
            </CardTitle>
          </CardHeader>
          <CardContent className="space-y-6">
            {/* Analysis Type Selection */}
            <div>
              <label className="text-sm font-medium mb-3 block">Analysis Type</label>
              <div className="grid md:grid-cols-2 gap-3">
                {documentTypes.map((type) => (
                  <div
                    key={type.value}
                    onClick={() => setAnalysisType(type.value)}
                    className={`p-4 border-2 rounded-lg cursor-pointer transition-all ${
                      analysisType === type.value
                        ? "border-purple-600 bg-purple-50"
                        : "border-gray-200 hover:border-purple-300"
                    }`}
                  >
                    <div className="font-medium text-sm">{type.label}</div>
                    <div className="text-xs text-gray-500 mt-1">{type.desc}</div>
                  </div>
                ))}
              </div>
            </div>

            {/* File Upload */}
            {!selectedFile ? (
              <div className="border-2 border-dashed border-gray-300 rounded-lg p-8 text-center hover:border-purple-400 transition-colors">
                <input
                  type="file"
                  accept=".pdf,.jpg,.jpeg,.png,.doc,.docx"
                  onChange={handleFileSelect}
                  className="hidden"
                  id="doc-upload"
                />
                <label htmlFor="doc-upload" className="cursor-pointer">
                  <div className="flex flex-col items-center gap-3">
                    <div className="flex gap-4">
                      <div className="w-16 h-16 bg-purple-100 rounded-xl flex items-center justify-center">
                        <Upload className="w-8 h-8 text-purple-600" />
                      </div>
                      <div className="w-16 h-16 bg-purple-100 rounded-xl flex items-center justify-center">
                        <FileText className="w-8 h-8 text-purple-600" />
                      </div>
                    </div>
                    <div>
                      <p className="text-gray-600 mb-2">Upload Document for AI Analysis</p>
                      <p className="text-sm text-gray-500">PDF, JPG, PNG, DOC, DOCX (max 10MB)</p>
                    </div>
                  </div>
                </label>
              </div>
            ) : (
              <div className="flex items-center justify-between p-4 bg-purple-50 border border-purple-200 rounded-lg">
                <div className="flex items-center gap-3">
                  {selectedFile.type.startsWith('image/') ? (
                    <ImageIcon className="w-8 h-8 text-purple-600" />
                  ) : (
                    <FileText className="w-8 h-8 text-purple-600" />
                  )}
                  <div>
                    <p className="font-medium text-gray-900">{selectedFile.name}</p>
                    <p className="text-sm text-gray-500">
                      {(selectedFile.size / 1024 / 1024).toFixed(2)} MB
                    </p>
                  </div>
                </div>
                <Button variant="ghost" size="icon" onClick={reset}>
                  <X className="w-5 h-5" />
                </Button>
              </div>
            )}

            <Button
              onClick={analyzeDocument}
              disabled={!selectedFile || isAnalyzing}
              className="w-full bg-purple-600 hover:bg-purple-700 text-white"
              size="lg"
            >
              {isAnalyzing ? (
                <>
                  <Loader2 className="w-5 h-5 mr-2 animate-spin" />
                  Analyzing Document...
                </>
              ) : (
                <>
                  <Sparkles className="w-5 h-5 mr-2" />
                  Analyze Document
                </>
              )}
            </Button>
          </CardContent>
        </Card>
      ) : (
        <Card>
          <CardHeader>
            <div className="flex items-center justify-between">
              <CardTitle className="flex items-center gap-2">
                <CheckCircle className="w-6 h-6 text-green-600" />
                Analysis Complete
              </CardTitle>
              <Button variant="outline" onClick={reset}>
                <X className="w-4 h-4 mr-2" />
                New Analysis
              </Button>
            </div>
          </CardHeader>
          <CardContent className="space-y-4">
            {analysisType === "comprehensive" && (
              <>
                <div className="p-4 bg-blue-50 border border-blue-200 rounded-lg">
                  <h4 className="font-semibold text-blue-900 mb-2">Document Type</h4>
                  <p className="text-blue-800">{analysis.document_type}</p>
                </div>

                <div className="p-4 bg-gray-50 rounded-lg">
                  <h4 className="font-semibold mb-2">Purpose</h4>
                  <p className="text-gray-700">{analysis.purpose}</p>
                </div>

                <div>
                  <h4 className="font-semibold mb-2">Key Points</h4>
                  <ul className="space-y-2">
                    {analysis.key_points?.map((point, i) => (
                      <li key={i} className="flex gap-2">
                        <CheckCircle className="w-4 h-4 text-green-600 mt-0.5" />
                        <span className="text-sm">{point}</span>
                      </li>
                    ))}
                  </ul>
                </div>

                <div className="p-4 bg-yellow-50 border border-yellow-200 rounded-lg">
                  <h4 className="font-semibold text-yellow-900 mb-2">Summary</h4>
                  <p className="text-yellow-800 text-sm">{analysis.summary}</p>
                </div>

                {analysis.concerns?.length > 0 && (
                  <div className="p-4 bg-red-50 border border-red-200 rounded-lg">
                    <h4 className="font-semibold text-red-900 mb-2 flex items-center gap-2">
                      <AlertCircle className="w-5 h-5" />
                      Concerns
                    </h4>
                    <ul className="space-y-1">
                      {analysis.concerns.map((concern, i) => (
                        <li key={i} className="text-sm text-red-800">‚Ä¢ {concern}</li>
                      ))}
                    </ul>
                  </div>
                )}
              </>
            )}

            {analysisType === "legal" && (
              <>
                <div>
                  <h4 className="font-semibold mb-2">Legal Terms</h4>
                  <div className="flex flex-wrap gap-2">
                    {analysis.legal_terms?.map((term, i) => (
                      <Badge key={i} variant="outline">{term}</Badge>
                    ))}
                  </div>
                </div>

                <div className="p-4 bg-blue-50 border border-blue-200 rounded-lg">
                  <h4 className="font-semibold text-blue-900 mb-2">Rights & Obligations</h4>
                  <ul className="space-y-1">
                    {analysis.rights_obligations?.map((item, i) => (
                      <li key={i} className="text-sm text-blue-800">‚Ä¢ {item}</li>
                    ))}
                  </ul>
                </div>

                {analysis.risks?.length > 0 && (
                  <div className="p-4 bg-red-50 border border-red-200 rounded-lg">
                    <h4 className="font-semibold text-red-900 mb-2">‚ö†Ô∏è Risks & Red Flags</h4>
                    <ul className="space-y-1">
                      {analysis.risks.map((risk, i) => (
                        <li key={i} className="text-sm text-red-800">‚Ä¢ {risk}</li>
                      ))}
                    </ul>
                  </div>
                )}

                <div className="p-4 bg-green-50 border border-green-200 rounded-lg">
                  <h4 className="font-semibold text-green-900 mb-2">‚úì Recommendations</h4>
                  <p className="text-sm text-green-800">{analysis.recommendations}</p>
                </div>
              </>
            )}

            {analysisType === "translation" && (
              <>
                <div className="p-4 bg-blue-50 rounded-lg">
                  <p className="text-sm text-blue-600 mb-1">Detected Language</p>
                  <p className="font-semibold">{analysis.original_language}</p>
                </div>
                <div className="p-4 bg-gray-50 rounded-lg">
                  <h4 className="font-semibold mb-2">Translated Content</h4>
                  <p className="text-sm whitespace-pre-wrap">{analysis.translated_text}</p>
                </div>
              </>
            )}
          </CardContent>
        </Card>
      )}
    </div>
  );
}
