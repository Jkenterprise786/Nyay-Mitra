import React, { useState, useEffect, useRef } from "react";
import { base44 } from "@/api/base44Client";
import { useMutation, useQuery, useQueryClient } from "@tanstack/react-query";
import { Button } from "@/components/ui/button";
import { Textarea } from "@/components/ui/textarea";
import { Send, Loader2, Bot, User as UserIcon, History, Plus, Trash2, MessageSquare, Paperclip, X, FileText, Image as ImageIcon } from "lucide-react";
import { toast } from "sonner";
import { format } from "date-fns";
import {
  Sheet,
  SheetContent,
  SheetDescription,
  SheetHeader,
  SheetTitle,
  SheetTrigger,
} from "@/components/ui/sheet";

export default function ChatInterface({ assistantType, assistantName, systemPrompt, user }) {
  const [messages, setMessages] = useState([]);
  const [input, setInput] = useState("");
  const [isLoading, setIsLoading] = useState(false);
  const [currentConversationId, setCurrentConversationId] = useState(null);
  const [historyOpen, setHistoryOpen] = useState(false);
  const [attachedFiles, setAttachedFiles] = useState([]);
  const [uploadingFiles, setUploadingFiles] = useState(false);
  const messagesEndRef = useRef(null);
  const fileInputRef = useRef(null);
  const queryClient = useQueryClient();

  // Load chat history
  const { data: conversations } = useQuery({
    queryKey: ['chat-history', user?.id, assistantType],
    queryFn: async () => {
      if (!user) return [];
      return await base44.entities.ChatHistory.filter({ 
        user_id: user.id, 
        assistant_type: assistantType 
      }, '-last_message_at');
    },
    initialData: [],
    enabled: !!user
  });

  const scrollToBottom = () => {
    messagesEndRef.current?.scrollIntoView({ behavior: "smooth" });
  };

  useEffect(() => {
    scrollToBottom();
  }, [messages]);

  const saveConversation = async (updatedMessages) => {
    if (!user || updatedMessages.length === 0) return;

    try {
      const userMessages = updatedMessages.filter(m => m.role === 'user');
      const conversationData = {
        user_id: user.id,
        assistant_type: assistantType,
        conversation_title: userMessages[0]?.content?.substring(0, 60) || "New Conversation",
        messages: updatedMessages,
        last_message_at: new Date().toISOString()
      };

      if (currentConversationId) {
        await base44.entities.ChatHistory.update(currentConversationId, conversationData);
      } else {
        const newConv = await base44.entities.ChatHistory.create(conversationData);
        setCurrentConversationId(newConv.id);
      }

      await queryClient.invalidateQueries(['chat-history', user?.id, assistantType]);
    } catch (error) {
      console.error("Failed to save conversation:", error);
      toast.error("Failed to save chat history");
    }
  };

  const handleFileSelect = async (e) => {
    const files = Array.from(e.target.files);
    if (files.length === 0) return;

    const oversizedFiles = files.filter(f => f.size > 10 * 1024 * 1024);
    if (oversizedFiles.length > 0) {
      toast.error("Some files exceed 10MB limit");
      return;
    }

    setUploadingFiles(true);
    try {
      const uploadedFiles = [];
      for (const file of files) {
        const { file_url } = await base44.integrations.Core.UploadFile({ file });
        uploadedFiles.push({
          name: file.name,
          url: file_url,
          type: file.type.startsWith('image/') ? 'image' : 'document'
        });
      }
      setAttachedFiles([...attachedFiles, ...uploadedFiles]);
      toast.success(`${uploadedFiles.length} file(s) attached`);
    } catch (error) {
      toast.error("Failed to upload files");
    } finally {
      setUploadingFiles(false);
    }
  };

  const removeAttachment = (index) => {
    setAttachedFiles(attachedFiles.filter((_, i) => i !== index));
  };

  const handleSend = async () => {
    if (!input.trim() && attachedFiles.length === 0) return;

    if (!user) {
      toast.error("Please login to use AI assistant");
      base44.auth.redirectToLogin();
      return;
    }

    const userMessage = {
      role: "user",
      content: input.trim() || "[Sent files]",
      timestamp: new Date().toISOString(),
      attachments: attachedFiles
    };

    const updatedMessages = [...messages, userMessage];
    setMessages(updatedMessages);
    setInput("");
    const currentAttachments = [...attachedFiles];
    setAttachedFiles([]);
    setIsLoading(true);

    try {
      const conversationHistory = updatedMessages.map(m => 
        `${m.role === 'user' ? 'User' : 'Assistant'}: ${m.content}`
      ).join('\n\n');

      let enhancedPrompt = `${systemPrompt}

Previous conversation:
${conversationHistory}

Respond to the latest user message in a helpful, clear, and concise manner.`;

      if (currentAttachments.length > 0) {
        enhancedPrompt += `\n\nThe user has attached ${currentAttachments.length} file(s). Please analyze these files and provide relevant insights, explanations, or assistance based on their content.`;
      }

      const llmParams = {
        prompt: enhancedPrompt
      };

      if (currentAttachments.length > 0) {
        llmParams.file_urls = currentAttachments.map(f => f.url);
      }

      const response = await base44.integrations.Core.InvokeLLM(llmParams);

      const assistantMessage = {
        role: "assistant",
        content: response,
        timestamp: new Date().toISOString()
      };

      const finalMessages = [...updatedMessages, assistantMessage];
      setMessages(finalMessages);
      
      // Save conversation automatically after each message
      await saveConversation(finalMessages);
    } catch (error) {
      console.error("Chat error:", error);
      toast.error("Failed to get response. Please try again.");
    } finally {
      setIsLoading(false);
    }
  };

  const loadConversation = (conversation) => {
    setCurrentConversationId(conversation.id);
    setMessages(conversation.messages || []);
    setHistoryOpen(false);
    toast.success("Conversation loaded!");
  };

  const startNewConversation = () => {
    setCurrentConversationId(null);
    setMessages([]);
    setAttachedFiles([]);
    setHistoryOpen(false);
    toast.success("Started new conversation");
  };

  const deleteConversation = async (conversationId, e) => {
    e.stopPropagation();
    try {
      await base44.entities.ChatHistory.delete(conversationId);
      if (currentConversationId === conversationId) {
        startNewConversation();
      }
      queryClient.invalidateQueries(['chat-history', user?.id, assistantType]);
      toast.success("Conversation deleted");
    } catch (error) {
      toast.error("Failed to delete conversation");
    }
  };

  return (
    <div className="flex flex-col h-full">
      {/* Header with History Button */}
      <div className="flex items-center justify-between p-4 border-b border-gray-200 bg-white">
        <div className="flex items-center gap-3">
          <div className="w-10 h-10 bg-gradient-to-br from-purple-400 to-purple-600 rounded-full flex items-center justify-center">
            <Bot className="w-6 h-6 text-white" />
          </div>
          <div>
            <h3 className="font-semibold text-gray-900">{assistantName}</h3>
            <p className="text-xs text-gray-500">AI-Powered Assistant with File Analysis</p>
          </div>
        </div>

        <div className="flex gap-2">
          <Button variant="outline" size="sm" onClick={startNewConversation}>
            <Plus className="w-4 h-4 mr-1" />
            New Chat
          </Button>
          
          <Sheet open={historyOpen} onOpenChange={setHistoryOpen}>
            <SheetTrigger asChild>
              <Button variant="outline" size="sm">
                <History className="w-4 h-4 mr-1" />
                History ({conversations.length})
              </Button>
            </SheetTrigger>
            <SheetContent className="w-full sm:max-w-md">
              <SheetHeader>
                <SheetTitle>Chat History</SheetTitle>
                <SheetDescription>
                  All conversations with {assistantName}
                </SheetDescription>
              </SheetHeader>
              
              <div className="mt-6 space-y-2 overflow-y-auto max-h-[calc(100vh-200px)]">
                {conversations.length > 0 ? (
                  conversations.map((conv) => (
                    <div
                      key={conv.id}
                      onClick={() => loadConversation(conv)}
                      className={`p-3 rounded-lg border cursor-pointer transition-colors hover:bg-gray-50 ${
                        currentConversationId === conv.id ? 'bg-purple-50 border-purple-200' : 'border-gray-200'
                      }`}
                    >
                      <div className="flex items-start justify-between gap-2">
                        <div className="flex-1 min-w-0">
                          <p className="font-medium text-sm text-gray-900 truncate">
                            {conv.conversation_title || "Untitled Conversation"}
                          </p>
                          <p className="text-xs text-gray-500 mt-1">
                            {format(new Date(conv.last_message_at), 'MMM d, h:mm a')}
                          </p>
                          <p className="text-xs text-gray-400 mt-1">
                            {conv.messages?.length || 0} messages
                          </p>
                        </div>
                        <Button
                          variant="ghost"
                          size="icon"
                          className="flex-shrink-0 h-8 w-8"
                          onClick={(e) => deleteConversation(conv.id, e)}
                        >
                          <Trash2 className="w-4 h-4 text-red-500" />
                        </Button>
                      </div>
                    </div>
                  ))
                ) : (
                  <div className="text-center py-8">
                    <MessageSquare className="w-12 h-12 text-gray-300 mx-auto mb-3" />
                    <p className="text-gray-500">No previous conversations</p>
                    <p className="text-xs text-gray-400 mt-2">Start chatting to save history</p>
                  </div>
                )}
              </div>
            </SheetContent>
          </Sheet>
        </div>
      </div>

      {/* Messages Area */}
      <div className="flex-1 overflow-y-auto p-4 space-y-4 bg-gray-50">
        {messages.length === 0 ? (
          <div className="h-full flex items-center justify-center">
            <div className="text-center max-w-md">
              <div className="w-16 h-16 bg-gradient-to-br from-purple-400 to-purple-600 rounded-full flex items-center justify-center mx-auto mb-4">
                <Bot className="w-8 h-8 text-white" />
              </div>
              <h3 className="text-xl font-semibold text-gray-900 mb-2">
                Welcome to {assistantName}!
              </h3>
              <p className="text-gray-600 mb-4">
                I'm here to help you. Ask me anything!
              </p>
              <div className="flex items-center justify-center gap-2 text-sm text-gray-500">
                <Paperclip className="w-4 h-4" />
                <span>You can attach images, documents, and forms for analysis</span>
              </div>
            </div>
          </div>
        ) : (
          <>
            {messages.map((message, index) => (
              <div
                key={index}
                className={`flex gap-3 ${message.role === 'user' ? 'justify-end' : 'justify-start'}`}
              >
                {message.role === 'assistant' && (
                  <div className="w-8 h-8 bg-gradient-to-br from-purple-400 to-purple-600 rounded-full flex items-center justify-center flex-shrink-0">
                    <Bot className="w-5 h-5 text-white" />
                  </div>
                )}
                
                <div
                  className={`max-w-[70%] rounded-2xl px-4 py-3 ${
                    message.role === 'user'
                      ? 'bg-purple-600 text-white'
                      : 'bg-white border border-gray-200 text-gray-900'
                  }`}
                >
                  <p className="text-sm whitespace-pre-wrap">{message.content}</p>
                  
                  {message.attachments && message.attachments.length > 0 && (
                    <div className="mt-2 space-y-1">
                      {message.attachments.map((file, i) => (
                        <div key={i} className={`flex items-center gap-2 text-xs p-2 rounded ${
                          message.role === 'user' ? 'bg-purple-500' : 'bg-gray-100'
                        }`}>
                          {file.type === 'image' ? <ImageIcon className="w-3 h-3" /> : <FileText className="w-3 h-3" />}
                          <span className="truncate">{file.name}</span>
                        </div>
                      ))}
                    </div>
                  )}
                  
                  <p className={`text-xs mt-1 ${message.role === 'user' ? 'text-purple-200' : 'text-gray-400'}`}>
                    {format(new Date(message.timestamp), 'h:mm a')}
                  </p>
                </div>

                {message.role === 'user' && (
                  <div className="w-8 h-8 bg-gradient-to-br from-orange-400 to-orange-600 rounded-full flex items-center justify-center flex-shrink-0">
                    <UserIcon className="w-5 h-5 text-white" />
                  </div>
                )}
              </div>
            ))}
            <div ref={messagesEndRef} />
          </>
        )}
      </div>

      {/* Input Area */}
      <div className="p-4 border-t border-gray-200 bg-white">
        {attachedFiles.length > 0 && (
          <div className="mb-3 flex flex-wrap gap-2">
            {attachedFiles.map((file, index) => (
              <div key={index} className="flex items-center gap-2 bg-purple-50 border border-purple-200 rounded-lg px-3 py-2">
                {file.type === 'image' ? <ImageIcon className="w-4 h-4 text-purple-600" /> : <FileText className="w-4 h-4 text-purple-600" />}
                <span className="text-sm text-purple-900 truncate max-w-[150px]">{file.name}</span>
                <button onClick={() => removeAttachment(index)} className="text-purple-600 hover:text-purple-800">
                  <X className="w-4 h-4" />
                </button>
              </div>
            ))}
          </div>
        )}

        <div className="flex gap-2">
          <input
            ref={fileInputRef}
            type="file"
            multiple
            accept=".pdf,.jpg,.jpeg,.png,.doc,.docx"
            onChange={handleFileSelect}
            className="hidden"
          />
          <Button
            variant="outline"
            size="icon"
            onClick={() => fileInputRef.current?.click()}
            disabled={uploadingFiles || isLoading}
            className="flex-shrink-0"
          >
            {uploadingFiles ? <Loader2 className="w-5 h-5 animate-spin" /> : <Paperclip className="w-5 h-5" />}
          </Button>
          <Textarea
            value={input}
            onChange={(e) => setInput(e.target.value)}
            onKeyDown={(e) => {
              if (e.key === 'Enter' && !e.shiftKey) {
                e.preventDefault();
                handleSend();
              }
            }}
            placeholder="Type your message or attach files..."
            className="resize-none"
            rows={2}
          />
          <Button
            onClick={handleSend}
            disabled={(!input.trim() && attachedFiles.length === 0) || isLoading}
            className="bg-purple-600 hover:bg-purple-700 text-white self-end"
          >
            {isLoading ? (
              <Loader2 className="w-5 h-5 animate-spin" />
            ) : (
              <Send className="w-5 h-5" />
            )}
          </Button>
        </div>
        <p className="text-xs text-gray-500 mt-2">
          Press Enter to send, Shift+Enter for new line â€¢ History saves automatically
        </p>
      </div>
    </div>
  );
}
